119 double Urban_Search_And_Rescue::m_calculate_yaw(geometry_msgs::Quaternion &follower_quat){
120  tf::Quaternion tf_quat;
121  tf::quaternionMsgToTF(follower_quat, tf_quat);
122  double roll{};
123  double pitch{};
124  double yaw{};
125  tf_quat.normalize(); //normalize the quaternion
126  tf::Matrix3x3(tf_quat).getRPY(roll, pitch, yaw); //calculates the roll, pitch and yaw from the quaternion
127  yaw = fmod(yaw, 2.0 * M_PI);
128  if (yaw <= 0.0){
129  yaw = yaw + 2.0 * M_PI; //limits the angle [0, 2*Pi]
130  }
131  return yaw;
